accountRepository
    <- login(credential)
    <- logout()
    -> message(msg)
    -> loggedIn(session)
    -> loggedOut()

deviceRepository
    queryHistory(deviceId, sensorId) -> points // caller owned
    <- connect(session)
    <- disconnect()
    <- send(deviceInfo, command)
    -> message(msg)
    -> attached(deviceMonitor) // callee owned
    -> detached(deviceId)

deviceMonitor
    <- update(deviceProperties) // forward update signal from upstream
    -> updated(deviceProperties)

sensorMonitor
    <- update(point) // forward update signal from upstream
